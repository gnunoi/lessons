\documentclass[border=12pt]{standalone}
\usepackage{tikz}
\usetikzlibrary{calc,intersections}
\begin{document}
\begin{tikzpicture}

\coordinate (O) at (0,0); % 设定原点O坐标
\coordinate (A) at (1,1); % 设定A点坐标
\coordinate (B) at (3,1); % 设定B点坐标
\coordinate (C) at (2,2); % 设定C点坐标

\draw (A) -- (B); % 画线AB
\draw (A) -- (C); % 画线AC

% $(起点)!放大系数!(终点)$，放大系数正负数均可
\coordinate (D) at ($(A)!2!(B)$); 
\coordinate (E) at ($(A)!-1!(B)$); 
\coordinate (F) at ($(A)!2!(C)$); 
\coordinate (G) at ($(A)!-0.5!(C)$); 
\foreach \p in {O,A,B,C,D,E,F,G} 
	\fill (\p) circle(1pt) node [below] {\textit{\p}};

\draw [red, densely dotted] (B) -- (D); % 画线BD
\draw [green!50!black, densely dotted] (A) -- (E); % 画线AE
\draw [red, densely dotted] (C) -- (F); % 画线BF
\draw [green!50!black, densely dotted] (A) -- (G); % 画线AG

\end{tikzpicture}
\end{document}
