\documentclass{standalone}
\usepackage{tikz}
\usetikzlibrary{calc,intersections}

\begin{document}

% 必须使用绝对坐标
\begin{tikzpicture}

\coordinate (A) at (0,0);
\coordinate (B) at (5,0);
\coordinate (C) at (0,3);
\coordinate (RO) at (4,4);

% rotate：旋转；around：围绕
% {0:(RO)}，围绕(RO)坐标，旋转0度
\begin{scope}[red, rotate around={0:(RO)}]
	\draw[blue] (0,0) -- (5,0) node [midway, below]{$a$};
	\draw[blue] (5,0) -- (0,3) node [midway, above]{$c$};
	\draw[blue] (0,3) -- (0,0) node [midway, left]{$b$};
\end{scope}

% {0:(RO)}，围绕(RO)坐标，旋转90度
\begin{scope}[red, rotate around={90:(RO)}]
    \draw (0,0) -- (5,0) node [midway, right]{$a$};
    \draw (5,0) -- (0,3) node [midway, left]{$c$};
    \draw (0,3) -- (0,0) node [midway, below]{$b$};
\end{scope}

% {0:(RO)}，围绕(RO)坐标，旋转180度
\begin{scope}[green!50!black, rotate around={180:(RO)}]
    \draw (0,0) -- (5,0) node [midway, above]{$a$};
    \draw (5,0) -- (0,3) node [midway, below]{$c$};
    \draw (0,3) -- (0,0) node [midway, right]{$b$};
\end{scope}

% {0:(RO)}，围绕(RO)坐标，旋转270度
\begin{scope}[orange, rotate around={270:(RO)}]
    \draw (0,0) -- (5,0) node [midway, left]{$a$};
    \draw (5,0) -- (0,3) node [midway, right]{$c$};
    \draw (0,3) -- (0,0) node [midway, above]{$b$};
\end{scope}


\end{tikzpicture}

\end{document}
